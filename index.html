name: Build APK on: push: branches: [ main, master ] workflow_dispatch: # 버튼 한 번으로 수동 빌드도 가능 jobs: build: runs-on: ubuntu-latest steps: # 1. 코드 체크아웃 - name: Checkout uses: actions/checkout@v4 # 2. JDK 17 세팅 - name: Set up JDK 17 uses: actions/setup-java@v4 with: java-version: '17' distribution: 'temurin' # 3. Gradle 캐시 (빌드 속도 대폭 향상) - name: Cache Gradle uses: actions/cache@v4 with: path: | ~/.gradle/caches ~/.gradle/wrapper key: ${{ runner.os }}-gradle-${{ hashFiles('**/*.gradle*', '**/libs.versions.toml') }} restore-keys: ${{ runner.os }}-gradle- # 4. gradlew 실행 권한 - name: Make gradlew executable run: chmod +x ./gradlew # 5. Debug APK 빌드 - name: Build Debug APK run: ./gradlew assembleDebug --no-daemon --stacktrace # 6. APK 위치 출력 - name: Show APK path run: find . -name "*.apk" -type f # 7. APK를 Artifacts로 업로드 (Actions 탭에서 다운로드 가능) - name: Upload Debug APK uses: actions/upload-artifact@v4 with: name: XWare-debug-${{ github.run_number }} path: app/build/outputs/apk/debug/app-debug.apk retention-days: 30 # ── 릴리즈 서명 APK (선택사항 — Secrets 설정 시 활성화) ── # Secrets 설정 방법: # GitHub repo → Settings → Secrets → New repository secret # KEYSTORE_BASE64 : base64로 인코딩한 .jks 파일 # KEY_ALIAS : 키 별칭 # KEY_PASSWORD : 키 비밀번호 # STORE_PASSWORD : 키스토어 비밀번호 - name: Sign Release APK if: ${{ secrets.KEYSTORE_BASE64 != '' }} run: | echo "${{ secrets.KEYSTORE_BASE64 }}" | base64 -d > keystore.jks ./gradlew assembleRelease \ -Pandroid.injected.signing.store.file=$(pwd)/keystore.jks \ -Pandroid.injected.signing.store.password=${{ secrets.STORE_PASSWORD }} \ -Pandroid.injected.signing.key.alias=${{ secrets.KEY_ALIAS }} \ -Pandroid.injected.signing.key.password=${{ secrets.KEY_PASSWORD }} \ --no-daemon - name: Upload Release APK if: ${{ secrets.KEYSTORE_BASE64 != '' }} uses: actions/upload-artifact@v4 with: name: XWare-release-${{ github.run_number }} path: app/build/outputs/apk/release/app-release.apk retention-days: 30
